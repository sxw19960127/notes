class Counter extends React.Component{
   constructor(props) {
      super(props);
      this.handleAddOne = this.handleAddOne.bind(this);
      this.handleMinusOne = this.handleMinusOne.bind(this);
      this.handleReset = this.handleReset.bind(this);
      this.state = {
         count: 0
      }
   }
   handleAddOne() {
      this.setState((prevState) => { //通过参数prevState获取到count值
         return{ //返回的这个对象仅仅会替换掉上面state中有关count的数值,不会影响其他数值发生改变
            count: prevState.count + 1
         }
      })
      console.log(this); //undefined
      console.log("加1")
   }
   handleMinusOne() {
      this.setState((prevState) => {
         return{
            count: prevState.count - 1
         }
      })
      console.log(this);
      console.log("减1")
   }
   handleReset() {
      this.setState(() => {
         return{
            count: 0
         }
      })
      console.log(this);
      console.log("重置");
   }
   render() {
      console.log(this);
      return(
         <div>
            <h1>Count: {this.state.count}</h1>
            {/* 绑定一个事件函数的时候,只是把函数的引用放到这里,实际上没有人去调用这个函数,函数是自执行的,所以会产生this指向丢失的问题,可以通过constructor去解决 */}
            <button onClick={this.handleAddOne}>+1</button>

class MyApp extends React.Component{
	//解决this指向丢失问题
   constructor(props) {
      super(props);
      this.handleRemoveAll = this.handleRemoveAll.bind(this);
      this.handlePickOption = this.handlePickOption.bind(this);
      this.handleAddOption = this.handleAddOption.bind(this);
      this.state = {
         options: ["张三","李四","王五","刘六"]
      }
   }
   
   //1.子组件需要调用在根组件中定义的数据,所以应该将处理函数定义在根组件中方便调用数据,再通过props传值给子组件
   handleAddOption(option) {
      // console.log(option);
      //验证input框里面的内容
      if(!option) {
         return "选项不能为空"
      }else if(this.state.options.includes(option)) {
         return "不能输入重复的选项"
      }else {
         this.setState((prevState) => { //渲染
            return{
               options: prevState.options.concat([option]) 
               //这里我们不能直接去改变prevState的值,只能使用concat拼接数组的形式去实现添加操作;
               //在不改变原数组的情况下重新生成一个数组
            }
         })
      } 
   }
   
   //2.
   handleRemoveAll() { 
      // console.log("删除所有")
      this.setState(() => {
         return{
            options: []
         }
      })
   }
   
   //3.
   handlePickOption() {
      const index = Math.floor(Math.random() * this.state.options.length);
      const option = this.state.options[index];
      alert(option);
   }
   
   //4.页面结构部分
