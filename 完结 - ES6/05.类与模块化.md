# 类

## 构造器

`ES5`中的仿类结构 --- 构造器：

```js
function Dog(name,age) {
	this.name = name;
	this.age = age;
}
// 给Dog构造器的原型对象上添加一个bark方法
Dog.prototype.bark = function() {
	console.log(`${this.name}`)
}
const lili = new Dog("lili",2);
const wangcai = new Dog("wangcai",1);
```

## 类

```js
class Dog{
    // 构造函数
	constructor(name,age) {
		this.name = name;
		this.age = age;
	}  
    // 不用加逗号
    // 给Dog原型对象,添加一个方法bark
    // 以下bark方法不能被遍历出来
	bark() {  
		console.log(`${this.name}`)
	}
}
const lili = new Dog("lili",2);
const wangcai = new Dog("wangcai",1);
```

### 注意事项

1. 类声明不会被提升 

2. 类声明下的代码都运行在严格模式下

3. 调用类必须使用`new`

4. 类中所有的方法都是不可枚举的

5. 类中的方法不能用`new`调用

6. 类的方法中重写类名会报错

```js
// ES5模拟实现类语法
// 1 6
const Dog = (function() {
    // 2
	"use strct";
	const Dog = function(name,age) {
        // 3
		if(!(this instanceof Dog)) {
			throw new Error("必须用new去调用方法")
		}
		this.name = name;
		this.age = age;
	}
	Object.defineProperty(Dog.prototype,"bark",{
		value: function() {
            // 5
			if(this instanceof Dog.prototype.bark) {
				throw new Error("不能使用new调用")
			}
			console.log(`${this.name}`)
		},
        // 4
		enumerable: false
	})
	return Dog;
})()
```

### `getter`和`setter`

设置读取属性。

```js
class Dog{
	constructor(name,age) {
		this.name = name;
		this.age = age;
	} 
	bark() { 
		console.log(`${this.name}`)
	}
    // 静态方法
    // 仅定义在当前函数中,可使用Dog.showInfo()调用
    // 通过实例调用lili.showInfo(),不能显示
	static showInfo() {
		console.log("这是一条狗")
	}
    // info也是一个属性
	get info() {
		return `我叫${this.name},我${this.age}了`
	}
    
    // nickName是一个属性
	set nickName(value) { // 设置值
		this.nick = value;
	}
	get nickName() { // 获取值
		return this.nick;
	}
}
const lili = new Dog('wangcai',3)
// console.log(lili.info)
lili.nickName = 'gouDan'
console.log(lili.nickName)
```

## 类的继承

### `ES5`中类的继承

```js
function Animal(name) {
	this.name = name;
	this.thirsty = 100;
	this.food = [];
}

Animal.prototype.drink = function() {
	return this.thirsty -= 10;
}
Animal.prototype.eat = function(item) {
	this.food.push(item);
}

function Dog(name, breed) {
    // 通过call改变Animal方法的this指向,指向当前函数对象,第二个参数传递进去name
	Animal.call(this,name);
	this.breed = breed;
}

Dog.prototype = Object.create(Animal.prototype); 

// 强制修改Dog的构造器为自身 
Dog.prototype.constructor = Dog; 

// Dog原型上有一个bark方法
Dog.prototype.bark = function() {
	console.log("狗在叫")
}

var wangcai = new Dog("wangcai", "中华田园犬"); 
console.log(wangcai)
```

### `ES6`中的继承

关键字`extend`实现继承，`super`方法执行一遍父级上的所有代码。

```js
class Animal{
	constructor(name) {
		this.name = name;
		this.thirsty = 100;
		this.food = []
	}
	drink() {
		return this.thirsty -= 10
	}
	eat(item) {
		this.food.push(item)
	}
}
class Dog extends Animal{
	constructor(name, breed) {
        // 执行一遍父级上的所有代码,必须在最前面
		super(name); 
		this.breed = breed
	}
	bark() {
		console.log("狗在叫")
	}
}
var wangcai = new Dog("wangcai", "中华田园犬"); 
console.log(wangcai)
```

### 静态方法的继承

```js
class Animal{
	constructor(name) {
		this.name = name;
		this.thirsty = 100;
		this.food = []
	}
	drink() {
		return this.thirsty -= 10
	}
	eat(item) {
		this.food.push(item)
	}
	static Info() {
		console.log("我是一个静态方法")
	}
}
class Dog extends Animal{
	constructor(name, breed) {
		super(name);  
		this.breed = breed
	}
	bark() {
		console.log("狗在叫")
	}
}
var wangcai = new Dog("wangcai", "中华田园犬"); 
// 直接调用静态方法
console.log(Dog.Info())
```

```js
// moviesList继承字Array,拥有Array所有的属性和方法
class moviesList extends Array{
    // 第二个参数的作用是将传递过来的参数全部转为数组形式
    constructor(name, ...movies) {
        // super走一遍Array中的所有代码
        // 传递的参数应该是一个个散列的数据,再次通过...的形式将数组拆分成散列的一个个数据
        super(...movies);
        this.name = name
    }
    add(movie) {
        this.push(movie)
    }
    // 按照评分高低进行筛选结果
    topStar(limit) {
        return this.sort((a,b) => {return a.star > b.star ? -1 : 1}).slice(0,limit)
    }
}

let movies = new moviesList('我最喜欢的电影', 
	{name: '绿皮书',star: 9},
	{name: '七龙珠',star: 3},
    {name: '黑猫警长',star: 5},
    {name: '卡布达',star: 6}
)
console.log(movies.topStar(2))
```



# 模块化

## 问题

`ES5`，我们通常使用立即执行函数去解决模块化问题，弊端：

- 并没有完全解决全局变量污染问题
- 需要考虑`script`加载顺序问题 

```js
(function() {
    function add(a,b) {
        return a + b
    }
    // add依旧存在全局变量污染的情况
    window.add = add
}())
```

## `ES6`模块化

导入`import`、导出`export`

```js
export var name = 'sxw'
```

```js
import {name} from './temp'
```

---

```html
<script type="module">
    import {add} from "./add.js";
</script>
```

```js
// add.js 
function add(a,b) {
	console.log(a+b)
}
export {add}  
```

---

```js
// 默认导出,在导入时的命名,可自定义取名 
export default function add(a,b) {
	console.log(a+b)
}
```

---

```js
// 当导入参数较多时
import * as obj from "./add.js"
console.log(obj)
```

---

```js
// 多变量暴露
var name = 'sxw'
var hobby = 'eat'

export {name,hobby};
```

---

```js
// 函数式输出
export function add(a,b) {
    return a + b;
}
```

---

```js
// 语义化输出
var a = 'shu'
var b = 'xiaowei'
var c = 'iu'
export {
	name as a,
    cname as b,
    love as iu
}
```

```js
import {name,cname,love} from './temp'
```

`export default`和`export`的区别：

- 前者在一个文件中只能有一个，后者可以存在多个

```js
export var a = 'sxw'
export function add(a,b) {
    return a + b
}
```

```js
import {a,add} from './temp'
```

---

```js
export default var a = 'sxw'
```

```js
// 默认导出的时候,我们在导入阶段可以自己定义名字
import iu from './temp'
```
